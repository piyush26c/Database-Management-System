CREATE OR REPLACE TRIGGER t
AFTER DELETE OR UPDATE ON clientmstr10_2
FOR EACH ROW
ENABLE
DECLARE
	opt audittrade10_2.opt%type;
BEGIN
	IF UPDATING THEN
		opt := 'update';
	ELSIF DELETING THEN
		opt := 'delete';
	END IF;
	INSERT INTO audittrade10_2 VALUES (:OLD.emp_name, :OLD.salary, opt);
END;
